property limit_to_size : false --if set to true, will only serch for images described in image_sizeproperty image_size : "large" --can be 'icon' 'small' 'medium' 'large'property limit_to_domain : false --if set to true, will search only in the domain described in search_domainproperty search_domain : "amazon.com" --search for images only in a certian domain/website, such as 'amazon.com' or 'artistdirect.com'property this_searchstring : "Ford Model T"property location : "~/Desktop/images/"-- you probably don't want to change anything below here-- property search_URL : "http://images.google.com/images?ie=ISO-8859-1&hl=en&btnG=Google+Search&q="property search_URL : "http://www.google.com/search?q="property parameters : "&tbm=isch&tbs=isz:ex,iszw:1920,iszh:1080"on run {brand, model}		set theFile to ((path to desktop) as text) & "images:" & brand & ":" & model & ".png"	set filestatus to false as boolean		tell application "Finder" to if exists theFile then set filestatus to true	if filestatus is true then		display notification "File " & model & " exists"	else		set the encoded_string to my urlencode(brand & " " & model, false, false)		if limit_to_size is true and limit_to_domain is true then			set the final_url to (search_URL & encoded_string & "&as_sitesearch=" & search_domain & "&imgsz=" & image_size & parameters)		else if limit_to_size is true and limit_to_domain is false then			set the final_url to (search_URL & encoded_string & "&imgsz=" & image_size)		else if limit_to_size is false and limit_to_domain is true then			set the final_url to (search_URL & encoded_string & "&as_sitesearch=" & search_domain)		else			set the final_url to (search_URL & encoded_string & parameters)		end if		tell me to access_website(final_url)				set theFolder to ((path to desktop) as text) & "images:" & brand		set status to false as boolean		--set p to path to desktop				tell application "Finder" to if exists theFolder then set status to true		if status is true then			display notification "Folder " & brand & " exists"		else			display notification "Created folder " & brand			do shell script ("mkdir -m 755 " & location & "/" & brand)		end if				delay 2				do shell script "date=$(date '+%Y%m%dT%H%M%S'); screencapture -x -R16,262,260,150 " & location & brand & "/" & urlencode(model, false, false) & ".png"				display notification "Got " & brand & " " & model				tell window 1 of application "Safari"			close (tabs where index = (get index of current tab))		end tell	end ifend runon urlencode(theText)	set theTextEnc to ""	repeat with eachChar in characters of theText		set useChar to eachChar		set eachCharNum to ASCII number of eachChar		if eachCharNum = 32 then			set useChar to "+"		else if (eachCharNum ≠ 42) and (eachCharNum ≠ 95) and (eachCharNum < 45 or eachCharNum > 46) and (eachCharNum < 48 or eachCharNum > 57) and (eachCharNum < 65 or eachCharNum > 90) and (eachCharNum < 97 or eachCharNum > 122) then			set firstDig to round (eachCharNum / 16) rounding down			set secondDig to eachCharNum mod 16			if firstDig > 9 then				set aNum to firstDig + 55				set firstDig to ASCII character aNum			end if			if secondDig > 9 then				set aNum to secondDig + 55				set secondDig to ASCII character aNum			end if			set numHex to ("%" & (firstDig as string) & (secondDig as string)) as string			set useChar to numHex		end if		set theTextEnc to theTextEnc & useChar as string	end repeat	return theTextEncend urlencodeon access_website(theURL)	try -- Jaguar default browser		tell application "Finder"			open location theURL		end tell	on error -- Panther and up default browser		tell application "System Events"			open location theURL		end tell	end tryend access_website
